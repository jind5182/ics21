/* Bubble sort: Pointer version */

void bubble_a(long *data, long count){
	long *point;
	long* last = data + count - 1;
  	long swap;
	while (last != data){
      	for (point = data; point != last; ++point){
          	if (*point > *(point+1)){
                swap = *(point+1);
              	*(point + 1) = *point;
              	*point = swap;
            }
        }
        last--;
    }
}
